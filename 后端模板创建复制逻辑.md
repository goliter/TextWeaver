# 复制流水线

## 模板创建流程图

```
用户点击"保存为模板"
        ↓
前端发送 POST /api/templates
        ↓
routes.py#create_template (API入口)
        ↓
验证工作流存在且属于当前用户
        ↓
crud.py#create_template (核心逻辑)
        ├──→ 创建模板主记录
        ├──→ _copy_nodes_to_template (复制节点)
        ├──→ _copy_edges_to_template (复制边)
        └──→ _copy_filesystem_to_template (复制文件系统)
        ↓
返回模板信息
```

---

## 详细流程解析

### 第一阶段：API 入口 (routes.py)

```python
@router.post("", response_model=schemas.WorkflowTemplateResponse, status_code=status.HTTP_201_CREATED)
def create_template(
    template_data: schemas.WorkflowTemplateCreate,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_active_user)
):
    """
    创建工作流模板 API 入口

    请求体示例：
    {
        "name": "PDF处理模板",
        "description": "自动提取PDF内容并分析",
        "tags": ["PDF", "文档处理"],
        "source_flow_id": 123  # 来源工作流ID
    }
    """
    # 1. 验证工作流存在且属于当前用户
    flow = db.query(Flow).filter(
        Flow.id == template_data.source_flow_id,
        Flow.user_id == current_user.id
    ).first()

    if not flow:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="工作流不存在或无权限"
        )

    # 2. 调用 CRUD 创建模板
    template = crud.create_template(db, template_data, current_user.id)
    return template
```

### 第二阶段：核心创建逻辑 (crud.py)

```python
def create_template(
    db: Session,
    template_data: schemas.WorkflowTemplateCreate,
    user_id: int
) -> models.WorkflowTemplate:
    """
    创建工作流模板 - 核心逻辑

    复制工作流的所有内容：节点、边、文件系统
    """

    # ========== 步骤1：创建模板主记录 ==========
    db_template = models.WorkflowTemplate(
        name=template_data.name,
        description=template_data.description,
        tags=template_data.tags,
        user_id=user_id,
        source_flow_id=template_data.source_flow_id,
        use_count=0,           # 使用次数初始为0
        is_public=False        # 默认私有
    )
    db.add(db_template)
    db.flush()  # 获取模板ID，但不提交事务

    # ========== 步骤2：复制工作流节点 ==========
    _copy_nodes_to_template(db, template_data.source_flow_id, db_template.id)

    # ========== 步骤3：复制工作流边 ==========
    _copy_edges_to_template(db, template_data.source_flow_id, db_template.id)

    # ========== 步骤4：复制文件系统 ==========
    _copy_filesystem_to_template(db, template_data.source_flow_id, db_template.id)

    # ========== 步骤5：提交事务 ==========
    db.commit()
    db.refresh(db_template)
    return db_template
```

### 第三阶段：复制节点

```python
def _copy_nodes_to_template(db: Session, flow_id: int, template_id: int) -> None:
    """复制工作流节点到模板"""

    # 1. 获取工作流的所有节点
    nodes = db.query(Node).filter(Node.flow_id == flow_id).all()

    # 2. 创建ID映射表：原始节点ID → 模板节点ID
    node_id_mapping = {}

    for node in nodes:
        # 3. 创建模板节点
        template_node = models.TemplateNode(
            template_id=template_id,
            node_type=node.node_type,           # 节点类型：start/ai/file_reader等
            name=node.name,                      # 节点名称
            position=node.position,              # 节点位置坐标
            data=node.data,                      # 节点配置数据（提示词、文件ID等）
            original_node_id=node.id             # 记录原始节点ID
        )
        db.add(template_node)
        db.flush()  # 获取新ID

        # 4. 记录ID映射关系
        node_id_mapping[node.id] = template_node.id

    # 5. 保存映射关系到模板（供后续边复制使用）
    db_template = db.query(models.WorkflowTemplate).filter(
        models.WorkflowTemplate.id == template_id
    ).first()
    if db_template:
        db_template.node_id_mapping = node_id_mapping
```

**关键设计**：

- `original_node_id`：保留原始节点ID，用于追溯
- `node_id_mapping`：记录新旧ID对应关系，用于复制边时更新引用

### 第四阶段：复制边

```python
def _copy_edges_to_template(db: Session, flow_id: int, template_id: int) -> None:
    """复制工作流边到模板"""

    # 1. 获取工作流的所有边
    edges = db.query(Edge).filter(Edge.flow_id == flow_id).all()

    # 2. 获取节点ID映射
    db_template = db.query(models.WorkflowTemplate).filter(
        models.WorkflowTemplate.id == template_id
    ).first()
    node_id_mapping = getattr(db_template, 'node_id_mapping', {})

    # 3. 复制每条边
    for edge in edges:
        # 4. 查找对应的模板节点ID
        source_template_node_id = node_id_mapping.get(edge.source_node_id)
        target_template_node_id = node_id_mapping.get(edge.target_node_id)

        # 5. 只有两端节点都存在才复制边
        if source_template_node_id and target_template_node_id:
            template_edge = models.TemplateEdge(
                template_id=template_id,
                source_node_id=source_template_node_id,      # 使用新的模板节点ID
                target_node_id=target_template_node_id,      # 使用新的模板节点ID
                source_handle=edge.source_handle,            # 连接点：bottom/right等
                target_handle=edge.target_handle,            # 连接点：top/left等
                original_edge_id=edge.id                     # 记录原始边ID
            )
            db.add(template_edge)
```

**关键设计**：

- 使用 `node_id_mapping` 更新边的引用关系
- 保留连接点信息（`source_handle`、`target_handle`）

### 第五阶段：复制文件系统（最复杂）

```python
def _copy_filesystem_to_template(db: Session, flow_id: int, template_id: int) -> None:
    """复制文件系统到模板"""

    # 1. 获取工作流的所有文件
    files = db.query(File).filter(File.flow_id == flow_id).all()

    # 2. 创建文件ID映射表
    file_id_mapping = {}

    # 3. 辅助函数：递归获取文件完整路径
    def get_file_path(file_obj: File) -> str:
        """递归获取文件的完整路径"""
        if file_obj.parent_id is None:
            return file_obj.name
        parent = db.query(File).filter(File.id == file_obj.parent_id).first()
        if parent:
            parent_path = get_file_path(parent)
            return f"{parent_path}/{file_obj.name}"
        return file_obj.name

    # ========== 第一轮：创建所有文件记录 ==========
    for file in files:
        template_file = models.TemplateFile(
            template_id=template_id,
            name=file.name,
            type=file.type.value if hasattr(file.type, 'value') else file.type,
            content=file.content,           # 复制文件内容
            size=file.size,
            path=get_file_path(file),       # 完整路径如：/folder1/folder2/file.txt
            original_file_id=file.id        # 记录原始文件ID
        )
        db.add(template_file)
        db.flush()  # 获取新ID

        file_id_mapping[file.id] = template_file.id

    # ========== 第二轮：更新父子关系 ==========
    for file in files:
        if file.parent_id and file.parent_id in file_id_mapping:
            template_file_id = file_id_mapping[file.id]
            parent_template_file_id = file_id_mapping[file.parent_id]

            # 更新父文件夹ID
            template_file = db.query(models.TemplateFile).filter(
                models.TemplateFile.id == template_file_id
            ).first()
            if template_file:
                template_file.parent_id = parent_template_file_id
```

**关键设计**：

- **两轮复制**：先创建所有文件，再更新父子关系
- 原因：子文件可能先于父文件夹被创建，需要等所有文件创建完成后再建立关系
- `path` 字段：保存完整路径，便于后续使用模板时重建文件结构

## 数据模型关系

```
WorkflowTemplate (模板主表)
    ├── TemplateNode (模板节点表) 1:N
    │       ├── node_type: 节点类型
    │       ├── data: 节点配置
    │       └── original_node_id: 原始节点ID
    │
    ├── TemplateEdge (模板边表) 1:N
    │       ├── source_node_id: 源节点ID（模板内）
    │       ├── target_node_id: 目标节点ID（模板内）
    │       └── source_handle/target_handle: 连接点
    │
    └── TemplateFile (模板文件表) 1:N
            ├── name: 文件名
            ├── content: 文件内容
            ├── path: 完整路径
            └── parent_id: 父文件夹ID（模板内）
```

# 创建模板

## 使用模板创建工作流流程图

```
用户点击"使用模板"
        ↓
前端发送 POST /api/templates/{template_id}/use
        ↓
routes.py#use_template (API入口)
        ↓
获取模板详情（包含节点、边、文件）
        ↓
1. 创建工作流
        ↓
2. 删除自动创建的根文件夹
        ↓
3. 复制文件系统（file_service.copy_template_filesystem_to_workflow）
        ↓
4. 创建节点（标准化节点类型）
        ↓
5. 创建边
        ↓
6. 更新节点文件引用（file_service.update_node_file_paths）
        ↓
7. 增加模板使用次数
        ↓
返回新工作流ID
```

---

## 详细流程解析

### 第一阶段：API 入口 (routes.py)

```python
@router.post("/{template_id}/use", response_model=schemas.UseTemplateResponse)
def use_template(
    template_id: int,
    use_data: schemas.UseTemplateRequest,  # {name, description}
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_active_user)
):
    """
    使用模板创建工作流 API 入口

    请求示例：
    POST /api/templates/123/use
    {
        "name": "我的PDF处理工作流",
        "description": "用于处理合同文档"
    }
    """

    # 1. 获取模板详情（包含节点、边、文件）
    template = crud.get_template_detail(db, template_id, current_user.id)

    if not template:
        raise HTTPException(status_code=404, detail="模板不存在或无权限")

    try:
        # ========== 步骤1：创建工作流 ==========
        flow_data = workflow_schemas.FlowCreate(
            name=use_data.name,
            description=use_data.description or template.description
        )
        new_flow = create_flow(db, flow_data, current_user.id)

        # ========== 步骤2：删除自动创建的根文件夹 ==========
        # 因为我们会从模板复制完整的文件系统
        auto_created_root = db.query(File).filter(
            File.flow_id == new_flow.id,
            File.parent_id.is_(None),
            File.type == 'folder'
        ).first()
        if auto_created_root:
            db.delete(auto_created_root)
            db.flush()

        # ========== 步骤3：复制文件系统 ==========
        file_id_mapping = file_service.copy_template_filesystem_to_workflow(
            db, template_id, new_flow.id
        )

        # ========== 步骤4：创建节点 ==========
        node_id_mapping = {}
        original_to_new_node_mapping = {}

        for template_node in template.nodes:
            # 标准化节点类型（关键！）
            node_type = template_node.node_type
            if node_type == 'fileReader':
                normalized_node_type = 'file_reader'
            elif node_type == 'fileWriter':
                normalized_node_type = 'file_writer'
            elif node_type == 'folderWriter':
                normalized_node_type = 'folder_writer'
            else:
                normalized_node_type = node_type

            node_data = workflow_schemas.NodeCreate(
                node_type=normalized_node_type,
                name=template_node.name,
                position=template_node.position,
                data=template_node.data
            )
            new_node = create_node(db, node_data, new_flow.id, current_user.id)
            node_id_mapping[template_node.id] = new_node.id
            original_to_new_node_mapping[template_node.original_node_id] = new_node.id

        # ========== 步骤5：创建边 ==========
        for template_edge in template.edges:
            source_node_id = node_id_mapping.get(template_edge.source_node_id)
            target_node_id = node_id_mapping.get(template_edge.target_node_id)

            if source_node_id and target_node_id:
                edge_data = workflow_schemas.EdgeCreate(...)
                create_edge(db, edge_data, new_flow.id, current_user.id)

        # ========== 步骤6：更新节点文件引用 ==========
        file_service.update_node_file_paths(
            db, template_id, new_flow.id,
            file_id_mapping, original_to_new_node_mapping
        )

        # ========== 步骤7：增加模板使用次数 ==========
        crud.increment_use_count(db, template_id)

        return {
            "flow_id": new_flow.id,
            "message": "工作流创建成功"
        }

    except Exception as e:
        db.rollback()
        raise HTTPException(status_code=500, detail=str(e))
```

### 第二阶段：复制文件系统 (file_service.py)

```python
def copy_template_filesystem_to_workflow(
    db: Session,
    template_id: int,
    flow_id: int
) -> Dict[int, int]:
    """
    将模板的文件系统复制到新工作流

    返回：文件ID映射 {template_file_id: new_file_id}
    """

    # 1. 获取模板的所有文件
    template_files = db.query(template_models.TemplateFile).filter(
        template_models.TemplateFile.template_id == template_id
    ).all()

    # 2. 创建文件ID映射表
    file_id_mapping = {}

    # ========== 第一轮：创建所有文件（不处理父子关系）==========
    for template_file in template_files:
        new_file = File(
            flow_id=flow_id,
            name=template_file.name,
            type=FileType(template_file.type),
            content=template_file.content,      # 复制文件内容
            size=template_file.size
        )
        db.add(new_file)
        db.flush()

        # 记录映射关系
        file_id_mapping[template_file.id] = new_file.id

    # ========== 第二轮：更新父子关系 ==========
    for template_file in template_files:
        if template_file.parent_id and template_file.parent_id in file_id_mapping:
            new_file_id = file_id_mapping[template_file.id]
            new_parent_id = file_id_mapping[template_file.parent_id]

            new_file = db.query(File).filter(File.id == new_file_id).first()
            if new_file:
                new_file.parent_id = new_parent_id

    db.commit()
    return file_id_mapping
```

**关键设计**：

- 两轮复制：先创建文件，再建立父子关系
- 返回 `file_id_mapping`：供后续更新节点引用使用

### 第三阶段：更新节点文件引用 (核心逻辑)

这是最关键的一步，将模板中的文件ID替换为新工作流的文件ID：

```python
def update_node_file_paths(
    db: Session,
    template_id: int,
    flow_id: int,
    file_id_mapping: Dict[int, int],
    original_to_new_node_mapping: Dict[int, int]
) -> None:
    """
    更新节点中的文件路径引用和prompt占位符

    问题：模板中保存的是原始工作流的文件ID
    解决：替换为新工作流的文件ID
    """

    # 1. 构建原始文件ID到新文件ID的映射
    # file_id_mapping: {template_file_id: new_file_id}
    # 需要转换为: {original_file_id: new_file_id}
    original_to_new_mapping = {}
    template_files = db.query(template_models.TemplateFile).filter(
        template_models.TemplateFile.template_id == template_id
    ).all()

    for template_file in template_files:
        if template_file.original_file_id and template_file.id in file_id_mapping:
            original_to_new_mapping[template_file.original_file_id] = file_id_mapping[template_file.id]

    # 2. 获取新工作流的所有节点
    nodes = db.query(Node).filter(Node.flow_id == flow_id).all()

    # 3. 更新每个节点的文件引用
    for node in nodes:
        updated = False
        node_data = node.data.copy() if node.data else {}

        # ========== 根据节点类型更新文件引用 ==========

        if node.node_type in ['file_reader', 'fileWriter']:
            # 文件读取/写入节点
            if 'fileId' in node_data:
                old_file_id = node_data.get('fileId')
                if old_file_id in original_to_new_mapping:
                    node_data['fileId'] = original_to_new_mapping[old_file_id]
                    updated = True

        elif node.node_type == 'folder_writer':
            # 文件夹写入节点
            if 'folderId' in node_data:
                old_folder_id = node_data.get('folderId')
                if old_folder_id in original_to_new_mapping:
                    node_data['folderId'] = original_to_new_mapping[old_folder_id]
                    updated = True

        # ========== 更新prompt中的占位符 ==========

        if 'prompt' in node_data and node_data['prompt']:
            prompt = node_data['prompt']
            original_prompt = prompt

            # 替换文件节点占位符 {file_66}
            # 注意：这里替换的是节点ID，不是文件ID！
            if original_to_new_node_mapping:
                for old_node_id, new_node_id in original_to_new_node_mapping.items():
                    prompt = prompt.replace(
                        f'{{file_{old_node_id}}}',
                        f'{{file_{new_node_id}}}'
                    )

            # 替换输入节点占位符 {input_104}
            if original_to_new_node_mapping:
                for old_node_id, new_node_id in original_to_new_node_mapping.items():
                    prompt = prompt.replace(
                        f'{{input_{old_node_id}}}',
                        f'{{input_{new_node_id}}}'
                    )

            # 替换输出节点占位符 {output_104}
            if original_to_new_node_mapping:
                for old_node_id, new_node_id in original_to_new_node_mapping.items():
                    prompt = prompt.replace(
                        f'{{output_{old_node_id}}}',
                        f'{{output_{new_node_id}}}'
                    )

            if prompt != original_prompt:
                node_data['prompt'] = prompt
                updated = True

        # 保存更新
        if updated:
            node.data = node_data

    db.commit()
```

## 关键问题：节点类型标准化

### 问题背景

前端和后端使用不同的节点类型命名：

| 前端           | 后端            |
| -------------- | --------------- |
| `fileReader`   | `file_reader`   |
| `fileWriter`   | `file_writer`   |
| `folderWriter` | `folder_writer` |

### 解决方案

```python
# 在创建节点时进行标准化转换
for template_node in template.nodes:
    node_type = template_node.node_type

    # 转换前端格式到后端标准格式
    if node_type == 'fileReader':
        normalized_node_type = 'file_reader'
    elif node_type == 'fileWriter':
        normalized_node_type = 'file_writer'
    elif node_type == 'folderWriter':
        normalized_node_type = 'folder_writer'
    else:
        normalized_node_type = node_type

    # 使用标准化后的类型创建节点
    node_data = workflow_schemas.NodeCreate(
        node_type=normalized_node_type,
        ...
    )
```

---

## 占位符替换详解

### 为什么需要替换占位符？

模板中的 prompt 可能包含对其他节点的引用：

```python
# 模板中的 prompt
prompt = """
请分析这个文件内容：
{{file_66}}

并根据以下要求处理：
{{input_104}}

输出结果到：
{{output_128}}
"""
```

创建新工作流后，节点ID都变了：

| 模板节点ID | 新工作流节点ID |
| ---------- | -------------- |
| 66         | 205            |
| 104        | 306            |
| 128        | 407            |

需要替换为：

```python
prompt = """
请分析这个文件内容：
{{file_205}}

并根据以下要求处理：
{{input_306}}

输出结果到：
{{output_407}}
"""
```

