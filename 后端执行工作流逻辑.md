## 整体流程图

```
前端点击"执行"
      ↓
POST /api/flows/{flow_id}/execute
      ↓
routes.py#execute_workflow (API入口)
      ↓
创建/获取 Execution 记录
      ↓
启动后台线程
      ↓
engine.py#execute_workflow_async (异步执行)
      ↓
engine.py#_execute_workflow_nodes (节点调度)
      ↓
engine.py#_execute_node (单节点执行)
      ↓
executors/#{node_type}#execute (具体节点逻辑)
      ↓
返回执行结果 (WebSocket实时推送)
```

## 第一阶段：API 入口 (routes.py)

### 代码解析

```python
@router.post("/{flow_id}/execute", response_model=schemas.WorkflowExecuteResponse)
def execute_workflow(
    flow_id: int,
    execute_request: Optional[schemas.WorkflowExecuteRequest] = None,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_active_user)
):
    """执行工作流 API 入口
    
    设计思路：
    1. 立即返回响应，不等待执行完成
    2. 在后台线程中执行工作流
    3. 通过 WebSocket 实时推送执行状态
    """
    
    # 1. 验证工作流存在且属于当前用户
    flow = crud.get_flow(db, flow_id=flow_id, user_id=current_user.id)
    if not flow:
        raise HTTPException(status_code=404, detail="Workflow not found")
    
    # 2. 获取输入数据（开始节点的输入）
    inputs = execute_request.inputs if execute_request else {}
    
    # 3. 检查是否有预创建的执行记录ID（用于WebSocket连接）
    execution_id = execute_request.execution_id if execute_request else None
    
    # 4. 创建执行引擎实例
    execution_engine = engine.ExecutionEngine(db)
    
    # 5. 同步创建执行记录（立即返回给前端）
    if execution_id:
        execution = crud.get_execution(db, execution_id=execution_id, user_id=current_user.id)
    else:
        execution = crud.create_execution(db, flow_id=flow_id, user_id=current_user.id)
    
    # 6. 在后台线程中异步执行工作流
    def run_workflow():
        try:
            execution_engine.execute_workflow_async(
                flow_id=flow_id, 
                user_id=current_user.id, 
                inputs=inputs, 
                execution_id=execution.id
            )
        except Exception as e:
            logging.error(f"Error executing workflow: {str(e)}")
    
    # 启动守护线程执行
    thread = threading.Thread(target=run_workflow, daemon=True)
    thread.start()
    
    # 7. 立即返回执行ID，不等待完成
    return {
        "execution_id": execution.id,
        "status": "running",
        "start_time": execution.start_time
    }
```

### 关键设计决策

| 设计 | 原因 |
|------|------|
| **立即返回** | HTTP 请求不能长时间挂起，避免超时 |
| **后台线程** | 工作流执行可能耗时很长，不能阻塞主线程 |
| **WebSocket 推送** | 实时通知前端执行进度和结果 |
| **预创建 execution_id** | 前端先建立 WebSocket 连接，再发送执行请求 |

## 第二阶段：执行引擎初始化 (engine.py)

### execute_workflow_async 方法

```python
def execute_workflow_async(self, flow_id: int, user_id: int, inputs: Dict[str, Any] = None, execution_id: int = None):
    """异步执行工作流
    
    执行流程：
    1. 发送开始状态（WebSocket）
    2. 获取节点和边
    3. 检测循环依赖
    4. 构建执行数据结构
    5. 查找开始节点
    6. 初始化数据流管理器
    7. 执行节点
    8. 更新最终状态
    """
    
    # 1. 发送执行开始状态到前端
    loop = get_or_create_event_loop()
    asyncio.run_coroutine_threadsafe(
        send_execution_status(execution.id, "start"), 
        loop
    )
    
    # 延迟 0.5 秒，给前端时间建立 WebSocket 连接
    time.sleep(0.5)
    
    # 2. 更新执行状态为运行中
    crud.update_execution_status(self.db, execution_id=execution.id, status="running")
    
    # 3. 获取工作流的所有节点和边
    nodes = crud.get_nodes_by_flow(self.db, flow_id=flow_id, user_id=user_id)
    edges = crud.get_edges_by_flow(self.db, flow_id=flow_id, user_id=user_id)
    
    # 4. 检测循环依赖（防止死循环）
    if self._detect_circular_dependency(nodes, edges):
        raise ValueError("Workflow contains circular dependencies")
    
    # 5. 构建节点映射表 {node_id: node}
    node_map = {node.id: node for node in nodes}
    
    # 6. 构建边映射表（按源节点分组）
    edges_by_source = {node.id: [] for node in nodes}
    for edge in edges:
        edges_by_source[edge.source_node_id].append(edge)
    
    # 7. 查找开始节点（必须有且只有一个）
    start_node = None
    for node in nodes:
        if node.node_type == "start":
            start_node = node
            break
    
    if not start_node:
        raise ValueError("Workflow must have a start node")
    
    # 8. 初始化数据流管理器
    data_flow = DataFlowManager(self.db, flow_id)
    
    # 设置开始节点的初始数据
    start_value = start_node.data.get("value", "")
    data_flow.set_node_output(start_node.id, "bottom", start_value)
    
    # 9. 执行工作流节点（核心方法）
    self._execute_workflow_nodes(
        start_node, 
        execution.id, 
        node_map, 
        edges_by_source, 
        data_flow
    )
    
    # 10. 发送执行成功状态
    asyncio.run_coroutine_threadsafe(
        send_execution_status(execution.id, "success"), 
        loop
    )
    
    # 11. 更新执行状态为成功
    crud.update_execution_status(
        self.db, 
        execution_id=execution.id, 
        status="success"
    )
```

## 第三阶段：节点执行调度 (核心算法)

### _execute_workflow_nodes 方法

```python
def _execute_workflow_nodes(
    self, 
    start_node: models.Node, 
    execution_id: int, 
    node_map: Dict[int, models.Node], 
    edges_by_source: Dict[int, List[models.Edge]],
    data_flow: DataFlowManager
):
    """节点执行调度器 - BFS + 拓扑排序
    
    核心逻辑：
    1. 使用队列实现 BFS 遍历
    2. 检查所有输入节点是否已执行
    3. 执行当前节点
    4. 将下游节点加入队列
    
    示例工作流：
    
    开始节点 ──→ AI节点 ──→ 选择节点 ──┬──→ 输出A
                                      └──→ 输出B
    
    执行顺序：
    1. 开始节点（入度为0，立即执行）
    2. AI节点（等待开始节点完成）
    3. 选择节点（等待AI节点完成，根据条件选择分支）
    4. 输出A 或 输出B（根据选择结果）
    """
    
    # 构建反向边映射（用于查找输入节点）
    edges_by_target = {node.id: [] for node in node_map.values()}
    for edge in sum(edges_by_source.values(), []):
        edges_by_target[edge.target_node_id].append(edge)
    
    # BFS 队列和已访问集合
    visited = set()
    queue = [start_node.id]
    
    # 选择节点的选择结果记录 {select_node_id: {target_node_id: True}}
    select_node_selections = {}
    
    while queue:
        current_node_id = queue.pop(0)
        
        # 跳过已执行的节点
        if current_node_id in visited:
            continue
        
        # ========== 关键：检查所有输入节点是否已执行 ==========
        all_inputs_executed = True
        for edge in edges_by_target.get(current_node_id, []):
            source_node_id = edge.source_node_id
            
            # 如果输入节点未执行，先执行输入节点
            if source_node_id not in visited:
                all_inputs_executed = False
                if source_node_id not in queue:
                    queue.append(source_node_id)
            else:
                # 特殊处理：如果输入是选择节点，检查当前节点是否被选中
                source_node = node_map.get(source_node_id)
                if source_node and source_node.node_type == "select":
                    if source_node_id in select_node_selections:
                        if current_node_id not in select_node_selections[source_node_id]:
                            all_inputs_executed = False
        
        # 只有所有输入都就绪，才执行当前节点
        if all_inputs_executed:
            visited.add(current_node_id)
            current_node = node_map.get(current_node_id)
            
            if not current_node:
                continue
            
            # ========== 执行节点 ==========
            outputs = self._execute_node(current_node, execution_id, data_flow)
            
            # ========== 处理输出，将下游节点加入队列 ==========
            if current_node.node_type == "select":
                # 选择节点：只选择一个分支
                self._handle_select_node(
                    current_node, outputs, edges_by_source,
                    queue, visited, select_node_selections, data_flow
                )
            else:
                # 普通节点：所有下游节点都加入队列
                for edge in edges_by_source.get(current_node_id, []):
                    target_node_id = edge.target_node_id
                    if target_node_id not in visited and target_node_id not in queue:
                        queue.append(target_node_id)
```

### 选择节点的特殊处理

```python
def _handle_select_node(
    self, 
    current_node, 
    outputs, 
    edges_by_source,
    queue, 
    visited, 
    select_node_selections, 
    data_flow
):
    """处理选择节点的输出
    
    选择节点根据AI的判断，只执行一个分支
    
    示例：
    选择节点判断"文件类型"
        ├── 如果是PDF → 执行PDF处理节点
        └── 如果是Word → 执行Word处理节点
    
    只会执行其中一个，不会两个都执行
    """
    added_targets = set()
    select_node_selections[current_node.id] = {}
    
    for edge in edges_by_source.get(current_node.id, []):
        target_node_id = str(edge.target_node_id)
        
        # 检查这个输出是否有数据（AI选择的结果）
        if target_node_id in outputs and outputs[target_node_id] is not None:
            if edge.target_node_id not in visited and edge.target_node_id not in queue:
                queue.append(edge.target_node_id)
                added_targets.add(edge.target_node_id)
                
                # 记录选择结果
                select_node_selections[current_node.id][edge.target_node_id] = True
                
                # 保存输出数据到数据流
                output_data = outputs[target_node_id]
                if isinstance(output_data, dict) and "handle" in output_data:
                    data_flow.set_node_output(
                        current_node.id, 
                        output_data["handle"], 
                        output_data["data"]
                    )
                
                # 只选择第一个匹配的，然后 break（关键！）
                break
```

### _execute_node 方法

```python
def _execute_node(
    self, 
    node: models.Node, 
    execution_id: int, 
    data_flow: DataFlowManager
) -> Dict[str, Any]:
    """执行单个节点
    
    流程：
    1. 创建执行日志
    2. 发送开始状态（WebSocket）
    3. 获取输入数据
    4. 获取节点执行器
    5. 执行节点逻辑
    6. 保存输出数据
    7. 发送完成状态（WebSocket）
    """
    
    # 1. 创建执行日志（用于记录输入输出）
    log = crud.create_execution_log(
        self.db, 
        execution_id=execution_id, 
        node_id=node.id
    )
    
    try:
        # 2. 发送 WebSocket：节点开始执行
        loop = get_or_create_event_loop()
        asyncio.run_coroutine_threadsafe(
            send_node_status(execution_id, node.id, "running"), 
            loop
        )
        
        # 3. 获取输入数据（从上游节点）
        inputs = data_flow.get_node_inputs(node.id)
        crud.update_execution_log(
            self.db, 
            log_id=log.id, 
            status="running", 
            input_data=inputs
        )
        
        # 4. 获取节点执行器（工厂模式）
        # node.node_type 可能是："start", "ai", "file_reader", "file_writer", "select", "output"
        executor = get_node_executor(self.db, node.node_type)
        
        # 5. 执行节点（调用具体执行器的 execute 方法）
        outputs = executor.execute(node, data_flow)
        
        # 6. 保存输出到数据流管理器（供下游节点使用）
        for handle, data in outputs.items():
            data_flow.set_node_output(node.id, handle, data)
        
        # 7. 发送 WebSocket：节点执行成功
        asyncio.run_coroutine_threadsafe(
            send_node_status(execution_id, node.id, "success", outputs), 
            loop
        )
        
        # 更新日志为成功
        crud.update_execution_log(
            self.db, 
            log_id=log.id, 
            status="success", 
            output_data=outputs
        )
        
        return outputs
        
    except Exception as e:
        # 发送错误状态
        asyncio.run_coroutine_threadsafe(
            send_node_status(execution_id, node.id, "error", None, str(e)), 
            loop
        )
        crud.update_execution_log(
            self.db, 
            log_id=log.id, 
            status="error", 
            error_message=str(e)
        )
        raise
```

```python
def get_node_executor(db: Session, node_type: str):
    """根据节点类型返回对应的执行器"""
    executors = {
        "start": StartNodeExecutor,
        "ai": AINodeExecutor,
        "file_reader": FileReaderExecutor,
        "file_writer": FileWriterExecutor,
        "select": SelectNodeExecutor,
        "output": OutputNodeExecutor,
    }
    
    executor_class = executors.get(node_type)
    if not executor_class:
        raise ValueError(f"Unknown node type: {node_type}")
    
    return executor_class(db)
```

## 第六阶段：数据流管理

### DataFlowManager 工作原理

```python
class DataFlowManager:
    """管理节点间的数据传递"""
    
    def __init__(self, db: Session, flow_id: int):
        self.db = db
        self.flow_id = flow_id
        # 存储节点输出：{node_id: {handle: data}}
        self._outputs = {}
    
    def set_node_output(self, node_id: int, handle: str, data: Any):
        """设置节点的输出数据"""
        if node_id not in self._outputs:
            self._outputs[node_id] = {}
        self._outputs[node_id][handle] = data
    
    def get_node_inputs(self, node_id: int) -> Dict[str, Any]:
        """获取节点的输入数据（从上游节点收集）"""
        inputs = {}
        
        # 获取所有指向当前节点的边
        edges = crud.get_edges_to_node(self.db, node_id)
        
        for edge in edges:
            source_node_id = edge.source_node_id
            source_handle = edge.source_handle or "bottom"
            target_handle = edge.target_handle or "left"
            
            # 从上游节点的输出中获取数据
            if source_node_id in self._outputs:
                source_outputs = self._outputs[source_node_id]
                if source_handle in source_outputs:
                    inputs[target_handle] = source_outputs[source_handle]
        
        return inputs
```

```
[开始节点] ──"hello"──→ [AI节点: "翻译为法语: {{input}}"] ──"bonjour"──→ [输出节点]
```

| 4 | 发送 WebSocket: node_running (开始节点) | {node_id: 1, status: "running"} |

## 关键设计总结

| 设计点 | 说明 |
|--------|------|
| **异步执行** | HTTP 立即返回，后台线程执行 |
| **WebSocket 实时推送** | 前端实时看到执行进度 |
| **BFS + 拓扑排序** | 确保按依赖顺序执行节点 |
| **数据流管理** | 节点间通过 DataFlowManager 传递数据 |
| **选择节点** | 实现条件分支，只执行选中分支 |
| **执行日志** | 记录每个节点的输入输出，便于调试 |
| **工厂模式** | 不同节点类型使用不同执行器 |
